package ru.lesson.l11;

public class Main {

    public static void main(String[] args) {

        if (args.length == 0) {
            System.out.println("Nothing to do");
            return;
        }

        String input = args[0];

        int wordCount = countWords(input);
        String longestWord = findLongestWord(input);

        System.out.println("Input was:");
        System.out.println(input);
        System.out.println();

        System.out.println("How much word in this phrase: ="+ wordCount + ";");
        System.out.println("The longest word: is -"+ longestWord + ";");
        System.out.println("The length of the longest word: ="+ longestWord.length() + ".");
    }

    // Метод подсчёта слов
    private static int countWords(String str) {
        String[] words = str.split("\\s+");					//работает со строкой как с последовательностью символов
        return words.length;
    }

    // Метод поиска самого длинного слова
    private static String findLongestWord(String str) {
        String[] words = str.split("\\s+");					//делит строку на части по шаблону (regex) ["Я", "прохожу", "курс", "Java"]

        String longest = words[0];

        for (String word : words) {
            if (word.length() > longest.length()) {
                longest = word;
            }
        }
        return longest;
    }
}

------------------ split --------------------------
String	объект-обёртка над char[]
split 	это высокоуровневая операция,
char	низкоуровневая единица данных
split 	это метод класса String:
	String[] split(String regex)
Он:
	НЕ работает с char напрямую
	работает со строкой как с последовательностью символов
	делит строку на части по шаблону (regex)
		Пример:
		String s = "Я прохожу курс Java";
		String[] words = s.split(" ");
Результат:
["Я", "прохожу", "курс", "Java"]
split 	работает внутри (упрощённо) Строка в Java внутри — это:
	[Я][ ][п][р][о][х][о][ж][у][ ][к][у][р][с]
split("\\s+") логически делает:
	идёт по символам
	проверяет каждый символ
	если символ соответствует regex
	заканчивает текущее слово
	собирает слова в массив String[]
⚠️ Но ты этого не видишь, потому что это:
	уже реализовано
	оптимизировано
	протестировано
========================= корректировака кода под вывод нескольких слов максимальной длины ==========
package ru.lesson.l11;

import java.util.ArrayList;
import java.util.List;

public class Main {

    public static void main(String[] args) {

        if (args.length == 0) {
            System.out.println("-----Nothing to do-----");
            return;
        }

        String input = args[0];

        String[] words = input.split("\\s+");

        int wordCount = words.length;
        int maxLength = findMaxWordLength(words);
        List<String> longestWords = findLongestWords(words, maxLength);

        System.out.println("Input was:");
        System.out.println(input);
        System.out.println();

        System.out.println("How much words in this phrase: = " + wordCount);
        System.out.println("The longest word length: = " + maxLength + " symbols");

        System.out.println("The longest word(s):");
        for (String word : longestWords) {
            System.out.println("- " + word);
        }
    }

    // 1️⃣ Максимальная длина слова
    private static int findMaxWordLength(String[] words) {
        int max = words[0].length();
        for (String word : words) {
            if (word.length() > max) {
                max = word.length();
            }
        }
        return max;
    }

    // 2️⃣ Все слова максимальной длины
    private static List<String> findLongestWords(String[] words, int maxLength) {
        List<String> result = new ArrayList<>();
        for (String word : words) {
            if (word.length() == maxLength) {
                result.add(word);
            }
        }
        return result;
    }
}
